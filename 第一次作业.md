#include <stdio.h>
#include <unistd.h>

int main() {
    while (1) {
        pid_t pid = fork();
        if (pid < 0) {
            printf("Fork failed.\n");
            return 1;
        } else if (pid == 0) {
           
            printf("Child process created.\n");
        } else {
            
            printf("Parent process. Parent PID: %d\n", pid);
        }
    }
    return 0;
}

            printf("Fork failed.\n");
            return 1;
        } else if (pid == 0) {
            
            printf("Child process. PID: %d, Parent PID: %d\n", getpid(), getppid());
